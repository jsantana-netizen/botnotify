# Notif Bot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Python bot that fetches quote and order data from the IGIQ API weekly and posts an 8-section behavioral analysis report to a Slack channel.

**Architecture:** Single Python script triggered by GitHub Actions cron every Monday at 8am UTC. Data is fetched from two IGIQ REST endpoints, processed in memory by isolated analyzer modules, assembled into a Slack-formatted message, and sent via Incoming Webhook. No database — all state is derived from API data on each run.

**Tech Stack:** Python 3.11+, `requests`, `pytest`, GitHub Actions, Slack Incoming Webhook

```python
import os
import time
import logging
from datetime import date, datetime
from typing import Optional
import requests
from analyzers import QuoteRecord, OrderRecord
import config

BASE_URL = "http://igiq-api.ignetworks.com/api/"
logger = logging.getLogger(__name__)


def _get_headers() -> dict:
    api_key = os.environ["IGIQ_API_KEY"]
    return {"Authorization": f"Bearer {api_key}"}  # adjust format after Step 1


def _get_with_retry(url: str, params: dict) -> dict:
    last_exc = None
    for attempt in range(config.API_MAX_RETRIES):
        try:
            resp = requests.get(
                url,
                headers=_get_headers(),
                params=params,
                timeout=config.API_TIMEOUT_SECONDS,
            )
            resp.raise_for_status()
            return resp.json()
        except requests.RequestException as e:
            last_exc = e
            if attempt < config.API_MAX_RETRIES - 1:
                time.sleep(2 ** attempt)
    raise last_exc


def _parse_quote(raw: dict) -> Optional[QuoteRecord]:
    project = raw.get("project")
    if not project:
        return None
    cluster = project.get("customer_cluster")
    if not cluster:
        return None

    dq = raw.get("date_quoted")
    dc = raw.get("date_created")
    return QuoteRecord(
        cq_id=raw["cq_id"],
        cluster_name=cluster["name"],
        cluster_slug=cluster["slug"],
        service_type=raw.get("related_service_type"),
        country=raw.get("related_country"),
        z_address=raw.get("z_address"),
        date_quoted=date.fromisoformat(dq) if dq else None,
        date_created=datetime.fromisoformat(dc).date() if dc else date.today(),
        status=raw.get("status"),
    )


def fetch_customer_quotes(
    due_date_after: str,
    due_date_before: Optional[str] = None,
) -> list[QuoteRecord]:
    url = BASE_URL + "pricing/customer-quotes/"  # update path after Step 1
    params = {"due_date_after": due_date_after}
    if due_date_before:
        params["due_date_before"] = due_date_before

    quotes = []
    page = 1
    while True:
        data = _get_with_retry(url, {**params, "page": page})
        for raw in data.get("results", []):
            q = _parse_quote(raw)
            if q:
                quotes.append(q)
        if not data.get("next"):
            break
        page += 1
    return quotes


def fetch_service_orders() -> list[OrderRecord]:
    url = BASE_URL + "accounting/service-orders/"
    orders = []
    page = 1
    while True:
        data = _get_with_retry(url, {"page": page})
        for raw in data.get("results", []):
            od = raw.get("order_date")
            if od:
                orders.append(OrderRecord(
                    order_id=raw["order_id"],
                    order_date=date.fromisoformat(od[:10]),
                ))
        if not data.get("next"):
            break
        page += 1
    return orders
```

# analyzers/no_purchase.py
from analyzers import QuoteRecord, OrderRecord


def find_no_purchase_clients(
    quotes: list[QuoteRecord],
    orders: list[OrderRecord],
) -> list[tuple[str, int]]:
    ordered_cluster_names = {o.order_id.lower() for o in orders}

    def has_order(cluster_name: str) -> bool:
        name_lower = cluster_name.lower()
        return any(name_lower in oid for oid in ordered_cluster_names)

    counts: dict[str, tuple[str, int]] = {}
    for q in quotes:
        slug = q.cluster_slug
        if slug not in counts:
            counts[slug] = (q.cluster_name, 0)
        counts[slug] = (counts[slug][0], counts[slug][1] + 1)

    result = [
        (name, count)
        for slug, (name, count) in counts.items()
        if not has_order(name)
    ]
    return sorted(result, key=lambda x: x[1], reverse=True)
```

# analyzers/reengaged.py
from datetime import date, timedelta
from analyzers import QuoteRecord
import config


def find_reengaged_clients(
    all_quotes: list[QuoteRecord],
    current_week_quotes: list[QuoteRecord],
    week_start: date,
) -> list[str]:
    cutoff = week_start - timedelta(days=config.REENGAGEMENT_DAYS)

    active_this_week = {q.cluster_slug for q in current_week_quotes}

    last_quote_before_week: dict[str, date] = {}
    for q in all_quotes:
        if q.cluster_slug in active_this_week:
            d = q.date_quoted or q.date_created
            if d < week_start:
                prev = last_quote_before_week.get(q.cluster_slug)
                if prev is None or d > prev:
                    last_quote_before_week[q.cluster_slug] = d

    reengaged = []
    for slug in active_this_week:
        last = last_quote_before_week.get(slug)
        if last is None or last < cutoff:
            name = next(q.cluster_name for q in current_week_quotes if q.cluster_slug == slug)
            reengaged.append(name)

    return sorted(reengaged)
```

# analyzers/churn_risk.py
from datetime import date
from analyzers import QuoteRecord
import config


def find_churn_risk_clients(
    quotes: list[QuoteRecord],
    today: date,
) -> list[tuple[str, int]]:
    last_seen: dict[str, tuple[str, date]] = {}
    for q in quotes:
        d = q.date_quoted or q.date_created
        prev = last_seen.get(q.cluster_slug)
        if prev is None or d > prev[1]:
            last_seen[q.cluster_slug] = (q.cluster_name, d)

    result = []
    for slug, (name, last_date) in last_seen.items():
        days_silent = (today - last_date).days
        if days_silent >= config.CHURN_RISK_DAYS:
            result.append((name, days_silent))

    return sorted(result, key=lambda x: x[1], reverse=True)
```

# analyzers/activity_spike.py
from collections import Counter
from analyzers import QuoteRecord
import config


def find_activity_spikes(
    current_week: list[QuoteRecord],
    previous_week: list[QuoteRecord],
) -> list[tuple[str, int, int]]:
    current_counts: Counter = Counter()
    cluster_names: dict[str, str] = {}
    for q in current_week:
        current_counts[q.cluster_slug] += 1
        cluster_names[q.cluster_slug] = q.cluster_name

    prev_counts: Counter = Counter()
    for q in previous_week:
        prev_counts[q.cluster_slug] += 1

    result = []
    for slug, current in current_counts.items():
        prev = prev_counts.get(slug, 0)
        if prev > 0 and current >= prev * config.ACTIVITY_SPIKE_MULTIPLIER:
            result.append((cluster_names[slug], current, prev))

    return sorted(result, key=lambda x: x[1], reverse=True)
```

# analyzers/top_states.py
import re
from collections import Counter
from typing import Optional
from analyzers import QuoteRecord
import config

_STATE_PATTERN = re.compile(r',\s*([A-Z]{2})\s*,\s*[\w\s]+,?\s*USA', re.IGNORECASE)

US_STATE_NAMES = {
    "AL": "Alabama", "AK": "Alaska", "AZ": "Arizona", "AR": "Arkansas",
    "CA": "California", "CO": "Colorado", "CT": "Connecticut", "DE": "Delaware",
    "FL": "Florida", "GA": "Georgia", "HI": "Hawaii", "ID": "Idaho",
    "IL": "Illinois", "IN": "Indiana", "IA": "Iowa", "KS": "Kansas",
    "KY": "Kentucky", "LA": "Louisiana", "ME": "Maine", "MD": "Maryland",
    "MA": "Massachusetts", "MI": "Michigan", "MN": "Minnesota", "MS": "Mississippi",
    "MO": "Missouri", "MT": "Montana", "NE": "Nebraska", "NV": "Nevada",
    "NH": "New Hampshire", "NJ": "New Jersey", "NM": "New Mexico", "NY": "New York",
    "NC": "North Carolina", "ND": "North Dakota", "OH": "Ohio", "OK": "Oklahoma",
    "OR": "Oregon", "PA": "Pennsylvania", "RI": "Rhode Island", "SC": "South Carolina",
    "SD": "South Dakota", "TN": "Tennessee", "TX": "Texas", "UT": "Utah",
    "VT": "Vermont", "VA": "Virginia", "WA": "Washington", "WV": "West Virginia",
    "WI": "Wisconsin", "WY": "Wyoming", "DC": "Washington D.C.",
}


def extract_state(z_address: Optional[str]) -> Optional[str]:
    if not z_address:
        return None
    match = _STATE_PATTERN.search(z_address)
    if not match:
        return None
    abbrev = match.group(1).upper()
    return abbrev if abbrev in US_STATE_NAMES else None


def get_top_states(
    quotes: list[QuoteRecord],
    n: int = config.TOP_N_STATES,
) -> list[tuple[str, str, int]]:
    counts: Counter = Counter()
    for q in quotes:
        if q.country != "United States":
            continue
        state = extract_state(q.z_address)
        if state:
            counts[state] += 1

    ranked = sorted(counts.items(), key=lambda x: (-x[1], x[0]))[:n]
    return [(abbrev, US_STATE_NAMES[abbrev], count) for abbrev, count in ranked]

# formatter.py
from datetime import date
from typing import Any


def _pct_change(current: int, previous: int) -> str:
    if previous == 0:
        return "N/A"
    pct = (current - previous) / previous * 100
    arrow = "↑" if pct >= 0 else "↓"
    return f"{arrow} {pct:+.1f}%"


def build_message(data: dict[str, Any]) -> str:
    ws = data["week_start"]
    we = data["week_end"]
    lines = [
        f"📊 *Reporte Semanal de Cotizaciones* — Semana del {ws.day} {ws.strftime('%b')} al {we.day} {we.strftime('%b %Y')}",
        "",
    ]

    SEP = "━━━━━━━━━━━━━━━━━━━━━━━━"

    if data["no_purchase"]:
        lines += [SEP, "🔴 *Clientes que cotizaron pero no compraron*", ""]
        for name, count in data["no_purchase"]:
            lines.append(f"• {name} — cotizó {count} servicios en los últimos 90 días y no ha ordenado ni un servicio nuevo.")
        lines.append("")

    if data["reengaged"]:
        lines += [SEP, "🟢 *Clientes que han vuelto a cotizar*", ""]
        for name in data["reengaged"]:
            lines.append(f"• {name}")
        lines.append("")

    if data["churn_risk"]:
        lines += [SEP, "⚠️ *Clientes en riesgo de churn*", ""]
        for name, days in data["churn_risk"]:
            lines.append(f"• {name} — última cotización hace {days} días.")
        lines.append("")

    if data["new_customers"]:
        lines += [SEP, "🆕 *Nuevos clientes que cotizaron por primera vez*", ""]
        for name in data["new_customers"]:
            lines.append(f"• {name}")
        lines.append("")

    if data["activity_spikes"]:
        lines += [SEP, "⚡ *Pico de actividad*", ""]
        for name, curr, prev in data["activity_spikes"]:
            lines.append(f"• {name} — {curr} cotizaciones esta semana (vs {prev} la semana pasada).")
        lines.append("")

    if data["top_services"]:
        lines += [SEP, "🏆 *Top tipos de servicio cotizados*", ""]
        for i, (stype, count) in enumerate(data["top_services"], 1):
            lines.append(f"{i}. {stype} — {count} cotizaciones")
        lines.append("")

    if data["top_states"]:
        lines += [SEP, "📍 *Top estados de USA cotizados*", ""]
        for i, (abbrev, name, count) in enumerate(data["top_states"], 1):
            lines.append(f"{i}. {name} ({abbrev}) — {count} cotizaciones")
        lines.append("")

    curr = data["current_week_total"]
    prev = data["prev_week_total"]
    lines += [
        SEP,
        "📈 *Variación de volumen global*",
        f"Total esta semana: {curr} cotizaciones",
        f"Total semana anterior: {prev} cotizaciones",
        f"{_pct_change(curr, prev)} vs semana anterior",
        "",
    ]

    return "\n".join(lines)
